#include <iostream>
	using namespace std;

class Animal {
	protected:
		string Color, Gender;
		int Size;
	public:
		string getColor() { return Color; }
		virtual void setColor(string c) { Color = c; }
		string getGender() { return Gender; }
		virtual void setGender(string g) { Gender = g; }
		int getSize() { return Size; }
		virtual void setSize(int s) { Size = s; }
		virtual void Eat() { cout << "The animal eats the food." << endl; }
		virtual void Sleep() { cout << "The animal goes to sleep." << endl; }
		virtual void Crap() { cout << "The animal... nevermind." << endl; }
};

class Dog : public Animal {
	private:
		string Breed;
	public:
		void setBreed(string b) { Breed = b; }
		string getBreed() { return Breed; }
		void setSize(int x) 
		{
			if (x > 200) {
				cout << "Your dog isn't that big! Make it smaller!" << endl;
				cout << "Please enter a new value: ";
				cin >> x;
				setSize(x);
			}
			Size = x;
		} 
		void Bark() { cout << "Woof!" << endl; }
		void Eat() { cout << "The dog eats the food." << endl; }
		void Crap() { cout << "You're cleaning that up." << endl; }
};

int main()
{
	Dog myDog;
	myDog.setColor("grey");
	myDog.setSize(300); 
	myDog.setGender("male");
	myDog.setBreed("Boxer");

	cout << "My Dog is a " << myDog.getGender() << " " << myDog.getColor() << " " << myDog.getBreed() << " and is " << myDog.getSize() << " inches." << endl;
	myDog.Bark();
	myDog.Eat();
	myDog.Crap();
	myDog.Sleep();
	
	return 0;
}

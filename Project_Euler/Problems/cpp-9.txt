// Find all Pythagorean triples with a max a or b values of x.
void findPythTrip( int x )
{
	for (int i=1; i<=x; i++)
	{
		for (int j=i+1; j<=x; i++)
		{
			double k = sqrt( (i*i) + (j*j) );
			if ( floor(k) == ( (i*i) + (j*j) ) )
			{
				cout << "Found Pythagorean Triple: " << i << "," << j << "," << k << endl;
			}
		}
	}
}

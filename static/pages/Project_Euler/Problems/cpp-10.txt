/*
	Check if the number is prime by seeing if it is divisble my any numbers other than itself.
	Note: There are much better algorithms than this; for instance, check it against all previous
	primes and squares of primes. Improve it later.
*/
bool checkIsPrime(int x)
{
	if (x==1) return false;  
	if ( (x==2) || (x==3) ) return true;
	for(int i=3; i<=x; i+=2)
	{
		if (i == x) return true;
		if (x%i == 0) return false;
	}
}

int main() {

	int checking = 3;
	int count = 2;

	while (checking < 15)
	{
		cout << "Currently checking: " << checking << endl;
		if (checkIsPrime(checking)) 
		{
			cout << "Found prime number " << count << ": " << checking << endl;
			count+=checking; 
			checking+=2;
		}
		else checking+=2;
	}

	cout << "The sum of all primes below 2,000,000 is : " << count << endl;

return 0;
}
